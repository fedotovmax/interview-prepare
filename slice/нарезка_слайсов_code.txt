package main

import "fmt"

func main() {
	a := []int{1, 2, 3, 4, 5, 6}

	b := a[2:5]

	fmt.Println("a", a, "a len=", len(a), "a cap=", cap(a)) // a [1 2 3 4 5 6] a len= 6 a cap= 6
	fmt.Println("b", b, "b len=", len(b), "b cap=", cap(b)) // b [3 4 5] b len= 3 b cap= 4

	a[3] = 11

	fmt.Println(a) // [1 2 3 11 5 6]
	fmt.Println(b) // [3 11 5]

	b[0] = 240

	fmt.Println(a) // [1 2 240 11 5 6]
	fmt.Println(b) // [240 11 5]
}
